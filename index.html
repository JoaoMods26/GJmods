GUI Universal Pro

Interfaz gráfica completa con múltiples funciones: ESP, Teleport, Speed, Jump Power y más. Compatible con la mayoría de juegos de Roblox.

🎯 Compatibilidad: Universal
🛡️ Estado: Seguro

Lua Script 
-- GUI Universal Pro by GJMODS
local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local TweenService = game:GetService("TweenService")

local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")

-- Crear GUI principal
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "GJModsGUI"
screenGui.Parent = playerGui

local mainFrame = Instance.new("Frame")
mainFrame.Size = UDim2.new(0, 400, 0, 500)

mainFrame.BorderSizePixel = 0

        dragging = true
        dragStart = input.Position

                                       startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

-- Título

ESP Universal

Utilidad

Sistema ESP completo que permite ver jugadores a través de las paredes con información detallada: nombre, distancia, salud y más.

👁️ Función: Visión ESP
🛡️ Detección: Baja
⚙️ Configuración: Avanzada
Lua Script 
-- ESP Universal by GJMODS
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")

local localPlayer = Players.LocalPlayer
local camera = workspace.CurrentCamera

local espEnabled = true
local espFolder = Instance.new("Folder")
espFolder.Name = "ESP_GJMODS"
espFolder.Parent = workspace

-- Configuración ESP
local espSettings = {
    showNames = true,
    showDistance = true,
    showHealth = true,
    maxDistance = 1000,
    textSize = 14
}

-- Función para crear ESP
local function createESP(player)
    if player == localPlayer then return end
    
    local character = player.Character or player.CharacterAdded:Wait()
    local humanoidRootPart = character:WaitForChild("HumanoidRootPart")
    local humanoid = character:WaitForChild("Humanoid")
    
    local billboard = Instance.new("BillboardGui")
    billboard.Name = player.Name .. "_ESP"
    billboard.StudsOffset = Vector3.new(0, 3, 0)
    billboard.Size = UDim2.new(0, 200, 0, 100)
    billboard.Parent = humanoidRootPart
    
    local nameLabel = Instance.new("TextLabel")
    nameLabel.Size = UDim2.new(1, 0, 0.4, 0)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Text = player.Name
    nameLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    nameLabel.TextStrokeTransparency = 0
    nameLabel.TextSize = espSettings.textSize
    nameLabel.Font = Enum.Font.SourceSansBold
    nameLabel.Parent = billboard
    
    local infoLabel = Instance.new("TextLabel")
    infoLabel.Size = UDim2.new(1, 0, 0.6, 0)
    infoLabel.Position = UDim2.new(0, 0, 0.4, 0)
    infoLabel.BackgroundTransparency = 1
    infoLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
    infoLabel.TextStrokeTransparency = 0
    infoLabel.TextSize = espSettings.textSize - 2
    infoLabel.Font = Enum.Font.SourceSans
    infoLabel.Parent = billboard
    
    -- Actualizar información
    local function updateESP()
        if not character.Parent then return end
        
        local distance = (localPlayer.Character.HumanoidRootPart.Position - humanoidRootPart.Position).Magnitude
        local health = humanoid.Health
        local maxHealth = humanoid.MaxHealth
        
        if distance > espSettings.maxDistance then
            billboard.Enabled = false
            return
        else
            billboard.Enabled = espEnabled
        end
        
        local infoText = ""
        if espSettings.showDistance then
            infoText = infoText .. string.format("Distancia: %.1f", distance) .. "\n"
        end
        if espSettings.showHealth then
            infoText = infoText .. string.format("Vida: %.0f/%.0f", health, maxHealth)
        end
        
        infoLabel.Text = infoText
        
        -- Cambiar color basado en la salud
        local healthPercent = health / maxHealth
        if healthPercent > 0.7 then
            nameLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
        elseif healthPercent > 0.3 then
            nameLabel.TextColor3 = Color3.fromRGB(255, 255, 0)
        else
            nameLabel.TextColor3 = Color3.fromRGB(255, 0, 0)
        end
    end
    
    RunService.Heartbeat:Connect(updateESP)
end

-- Aplicar ESP a todos los jugadores
for _, player in pairs(Players:GetPlayers()) do
    if player.Character then
        createESP(player)
    end
    player.CharacterAdded:Connect(function()
        createESP(player)
    end)
end

Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function()
        createESP(player)
    end)
end)

-- Toggle ESP con tecla
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Enum.KeyCode.E then
        espEnabled = not espEnabled
        print("ESP:", espEnabled and "Activado" or "Desactivado")
    end
end)

print("ESP Universal cargado! Presiona 'E' para activar/desactivar")

Speed & Jump Enhancer

Utilidad

Modificador de velocidad y salto con controles de teclado. Incluye modo sigiloso y configuración personalizable para evitar detecciones.

🏃 Velocidad: Configurable
🦘 Salto: Mejorado
🔧 Controles: Personalizables
Lua Script 
-- Speed & Jump Enhancer by GJMODS
local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:WaitForChild("Humanoid")

-- Configuración
local config = {
    normalSpeed = 16,
    enhancedSpeed = 50,
    normalJump = 50,
    enhancedJump = 120,
    stealthMode = false
}

local speedEnabled = false
local jumpEnabled = false

-- Función para actualizar velocidad
local function updateSpeed()
    if speedEnabled then
        humanoid.WalkSpeed = config.stealthMode and 25 or config.enhancedSpeed
    else
        humanoid.WalkSpeed = config.normalSpeed
    end
end

-- Función para actualizar salto
local function updateJump()
    if jumpEnabled then
        humanoid.JumpPower = config.enhancedJump
    else
        humanoid.JumpPower = config.normalJump
    end
end

-- Sistema de notificaciones
local function notify(message)
    local gui = Instance.new("ScreenGui")
    gui.Parent = player:WaitForChild("PlayerGui")
    
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0, 300, 0, 60)
    frame.Position = UDim2.new(1, -320, 0, 20)
    frame.BackgroundColor3 = Color3.fromRGB(30, 30, 46)
    frame.BorderSizePixel = 0
    frame.Parent = gui
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 8)
    corner.Parent = frame
    
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, -20, 1, 0)
    label.Position = UDim2.new(0, 10, 0, 0)
    label.BackgroundTransparency = 1
    label.Text = message
    label.TextColor3 = Color3.white
    label.TextScaled = true
    label.Font = Enum.Font.SourceSansBold
    label.Parent = frame
    
    -- Animación de entrada
    frame:TweenPosition(UDim2.new(1, -320, 0, 20), "Out", "Quad", 0.3, true)
    
    -- Eliminar después de 3 segundos
    game:GetService("Debris"):AddItem(gui, 3)
    
    wait(2.5)
    frame:TweenPosition(UDim2.new(1, 20, 0, 20), "Out", "Quad", 0.3, true)
end

-- Controles de teclado
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Enum.KeyCode.LeftShift then
        speedEnabled = not speedEnabled
        updateSpeed()
        notify("Speed: " .. (speedEnabled and "ON" or "OFF"))
        
    elseif input.KeyCode == Enum.KeyCode.LeftControl then
        jumpEnabled = not jumpEnabled
        updateJump()
        notify("Jump Power: " .. (jumpEnabled and "ON" or "OFF"))
        
    elseif input.KeyCode == Enum.KeyCode.F then
        config.stealthMode = not config.stealthMode
        if speedEnabled then updateSpeed() end
        notify("Stealth Mode: " .. (config.stealthMode and "ON" or "OFF"))
    end
end)

-- Reconectar cuando el jugador respawnea
player.CharacterAdded:Connect(function(newCharacter)
    character = newCharacter
    humanoid = character:WaitForChild("Humanoid")
    
    -- Aplicar configuraciones si estaban activas
    if speedEnabled then updateSpeed() end
    if jumpEnabled then updateJump() end
end)

print("Speed & Jump Enhancer cargado!")
print("Controles:")
print("LeftShift = Toggle Speed")
print("LeftControl = Toggle Jump")
print("F = Toggle Stealth Mode")

Auto Farm Universal

Juego Específico

Sistema de auto farm inteligente que detecta objetos coleccionables y los recolecta automáticamente. Incluye configuración de distancia y filtros personalizables.

🤖 Modo: Automático
📏 Distancia: Configurable
🎯 Precisión: Alta
Lua Script 
-- Auto Farm Universal by GJMODS
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local TweenService = game:GetService("TweenService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoidRootPart = character:WaitForChild("HumanoidRootPart")

-- Configuración del Auto Farm
local farmConfig = {
    enabled = false,
    maxDistance = 100,
    speed = 50,
    collectDelay = 0.1,
    targetObjects = {"Coin", "Cash", "Money", "Gem", "Crystal", "Orb"}
}

-- Función para encontrar objetos coleccionables
local function findCollectibles()
    local collectibles = {}
    
    for _, obj in pairs(workspace:GetDescendants()) do
        if obj:IsA("BasePart") or obj:IsA("Model") then
            for _, targetName in pairs(farmConfig.targetObjects) do
                if string.find(obj.Name:lower(), targetName:lower()) then
                    local distance = (humanoidRootPart.Position - obj.Position).Magnitude
                    if distance <= farmConfig.maxDistance then
                        table.insert(collectibles, {object = obj, distance = distance})
                    end
                end
            end
        end
    end
    
    -- Ordenar por distancia
    table.sort(collectibles, function(a, b)
        return a.distance < b.distance
    end)
    
    return collectibles
end

-- Función para teletransportarse a un objeto
local function teleportTo(position)
    local tween = TweenService:Create(
        humanoidRootPart,
        TweenInfo.new(0.5, Enum.EasingStyle.Linear),
        {CFrame = CFrame.new(position)}
    )
    tween:Play()
    return tween
end

-- Sistema de auto farm
local farmConnection
local function startAutoFarm()
    farmConnection = RunService.Heartbeat:Connect(function()
        if not farmConfig.enabled then return end
        
        local collectibles = findCollectibles()
        
        for _, item in pairs(collectibles) do
            if not farmConfig.enabled then break end
            
            local obj = item.object
            if obj and obj.Parent then
                -- Teletransportarse al objeto
                teleportTo(obj.Position)
                wait(farmConfig.collectDelay)
                
                -- Intentar tocar/recolectar el objeto
                if obj:IsA("BasePart") then
                    firetouchinterest(humanoidRootPart, obj, 0)
                    firetouchinterest(humanoidRootPart, obj, 1)
                end
                
                wait(farmConfig.collectDelay)
            end
        end
        
        wait(0.1)
    end)
end

-- Función para detener el auto farm
local function stopAutoFarm()
    if farmConnection then
        farmConnection:Disconnect()
        farmConnection = nil
    end
end

-- Sistema de notificaciones
local function notify(message, color)
    color = color or Color3.fromRGB(0, 255, 0)
    
    local gui = Instance.new("ScreenGui")
    gui.Parent = player:WaitForChild("PlayerGui")
    
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0, 300, 0, 50)
    frame.Position = UDim2.new(1, -320, 0, 20)
    frame.BackgroundColor3 = Color3.fromRGB(30, 30, 46)
    frame.BorderSizePixel = 0
    frame.Parent = gui
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 8)
    corner.Parent = frame
    
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, -20, 1, 0)
    label.Position = UDim2.new(0, 10, 0, 0)
    label.BackgroundTransparency = 1
    label.Text = message
    label.TextColor3 = color
    label.TextScaled = true
    label.Font = Enum.Font.SourceSansBold
    label.Parent = frame
    
    game:GetService("Debris"):AddItem(gui, 3)
end

-- Controles de teclado
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Enum.KeyCode.F1 then
        farmConfig.enabled = not farmConfig.enabled
        
        if farmConfig.enabled then
            startAutoFarm()
            notify("Auto Farm: ACTIVADO", Color3.fromRGB(0, 255, 0))
        else
            stopAutoFarm()
            notify("Auto Farm: DESACTIVADO", Color3.fromRGB(255, 100, 100))
        end
    end
end)

-- Reconectar cuando el jugador respawnea
player.CharacterAdded:Connect(function(newCharacter)
    character = newCharacter
    humanoidRootPart = character:WaitForChild("HumanoidRootPart")
    
    if farmConfig.enabled then
        stopAutoFarm()
        wait(1)
        startAutoFarm()
    end
end)

print("Auto Farm Universal cargado!")
print("Presiona F1 para activar/desactivar")
print("Objetos objetivo:", table.concat(farmConfig.targetObjects, ", "))
